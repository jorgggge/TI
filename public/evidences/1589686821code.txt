using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Prolog
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine(" -- Ingresa el precio de los burritos: ");
            decimal Precio = Decimal.Parse(Console.ReadLine()); 
            Console.WriteLine("----------------- Tabla de Precios ----------------------");

            var Tabla_Precios = GetPercios(Precio);
            int x = 1;
            foreach (var _Precio in Tabla_Precios) {

                if (x <= 15) Console.WriteLine(x + " * " + Precio + " = " + _Precio);
                else break;

                x++;
            }

            Console.ReadKey();
        }

        static IEnumerable<decimal> GetPercios(decimal Precio_Actual) {

            yield return Precio_Actual;
            decimal Aux = Precio_Actual;

            while (true) {
                Aux = Precio_Actual + Aux;
                yield return Aux;
            }
        }

    }
}
